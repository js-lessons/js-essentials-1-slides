Специальные числа


var x = 2 / 0;          // Infinity
var y = -2 / 0;         // -Infinity

var z = "apple" / 3     // NaN

typeof x                // "number"
typeof y                // "number"
typeof z                // "number"

isNaN(z)                // true
isNaN(x + y)            // true

Логические операторы && и ||


var role = null || "user"        // role = "user"
var role = "admin" || "user"     // role = "admin"

var result = undefined &&  10    // result = undefined
var result = 100 &&  10          // result = 100
